public class Main {
    public static int imperative_sum(List<Integer> xs) {
        int sum = 0;
        for (int x : xs) {
            sum += x;
        }
        return sum;
    }

    public static int functional_sum(List<Integer> xs) {
        if (xs.isEmpty()) {
            return 0;
        }

        return xs[0] + functional_sum(xs.sublist(1, xs.size()));
    }
}
\end{lstlisting}
